    """
    Body class with Massoud parameterization

    Uses calls to the Massoud Python wrapper

    def __init__(self,name,id=0,group=None,boundary=0,fun3d=True,motion_type='deform',
                 ndv=0,  aero_gp_file='',  struct_gp_file='', dv_file='',  usd_file='',comm=None):
        """

        Parameters
        ----------
        name: str
            name of the body
        id: int
            id number of the body
        group: int
            group number for coupling of variables
        boundary: int
            FUN3D boundary number associated with the body
        fun3d: bool
            whether or not you are using FUN3D. If true, the body class will auto-populate 'rigid_motion' required by FUN3D
        motion_type:
            the type of motion the body is undergoing. Possible options: 'deform','rigid','deform+rigid'
        ndv: int
            number of shape design variables in the MASSOUD parameterization
        aero_gp_file: str
            name of the file with the MASSOUD parameterized aerodynamic surface mesh
        struct_gp_file: str
            name of the file with the MASSOUD parameterized structural mesh
        dv_file: str
            name of the MASSOUD dv_file
        usd_file: str
            name of the MASSOUD usd_file
        comm: MPI comm
            the FUNtoFEM communicator

        See Also
        --------
        :mod:`body` : MassoudBody inherits from Body
        """

        from variable import Variable as dv

        super(MassoudBody,self).__init__(name,id,group,boundary,fun3d,motion_type)

        self.shape = True

        self.ndv = ndv
        self.m = Massoud()
        self.aero_gp_file   = aero_gp_file
        self.struct_gp_file = struct_gp_file
        self.dv_file        = dv_file
        self.usd_file       = usd_file

        self.aero_X      = None
        self.aero_id     = None
        self.aero_nnodes = None
        self.aero_sd     = None

        self.struct_X      = None
        self.struct_id     = None
        self.struct_nnodes = None
        self.struct_sd     = None

        self.comm = comm

        for v in xrange(ndv):
            self.add_variable('shape',dv('shape '+str(v),active=False,id=v+1))


[docs]    def initialize_shape_parameterization(self):
        """
        **[driver call]**
        Create the map of structural node locations since TACS doesn't give us global id numbers

        """

        if not self.struct_id:
            self.struct_id  = range(self.struct_nnodes)

            # These are the initial coordinates given by the solver
            struct_X = self.struct_X.copy()
            self.struct_id = self._form_struct_map(struct_X)



[docs]    def update_shape(self,complex_run=False):
        """
        **[driver call]**
        perturbs the shape of the body based on the body's shape variables and update the meshes

        Parameters
        ----------
        complex_run: bool
            whether or not the run is complex mode

        """

        os.chdir("./Rubberize")

        self._massoud_update()

        if complex_run:
            self._complex_perturb()

        os.chdir("..")

        return



[docs]    def shape_derivative(self,scenario,offset):
        """
        **[driver call]**
        Calculates the shape derivative given the coordinate derivatives

        Parameters
        ----------
        scenario: scenario object
            The current scenario
        offset: int
            function offset number
        """
        for func in xrange(len(scenario.functions)):
            for var in xrange(self.ndv):
                self.derivatives['shape'][offset+func][var] = 0.0

        for pt in xrange(self.aero_nnodes):
            for func in xrange(len(scenario.functions)):
                for var in xrange(self.ndv):
                    self.derivatives['shape'][offset+func][var] += self.aero_shape_term[3*pt  ,func] * self.aero_sd[pt*self.ndv*3+var*3  ]
                    self.derivatives['shape'][offset+func][var] += self.aero_shape_term[3*pt+1,func] * self.aero_sd[pt*self.ndv*3+var*3+1]
                    self.derivatives['shape'][offset+func][var] += self.aero_shape_term[3*pt+2,func] * self.aero_sd[pt*self.ndv*3+var*3+2]

        for pt in xrange(self.struct_nnodes):
            for func in xrange(len(scenario.functions)):
                for var in xrange(self.ndv):
                    self.derivatives['shape'][offset+func][var] += self.struct_shape_term[3*pt  ,func] * self.struct_sd[pt*self.ndv*3+var*3  ]
                    self.derivatives['shape'][offset+func][var] += self.struct_shape_term[3*pt+1,func] * self.struct_sd[pt*self.ndv*3+var*3+1]
                    self.derivatives['shape'][offset+func][var] += self.struct_shape_term[3*pt+2,func] * self.struct_sd[pt*self.ndv*3+var*3+2]

        # Get the contributions from all the processors
        for func in xrange(len(scenario.functions)):
            for var in xrange(self.ndv):
                deriv = np.array(self.derivatives['shape'][offset+func][var])
                self.derivatives['shape'][offset+func][var] = self.comm.allreduce(deriv)


    def _massoud_update(self,dv=None,create_map=False):
        """
        Perturbs the surface and update coordinate sensitivities

        Parameters
        ----------
        dv: list of float
            new design variable values

        """

        # form a list of the design variables
        if type(dv)==type(None):
            dv = []
            for var in self.variables['shape']:
                if type(var.value) == complex or type(var.value) == np.complex128:
                    dv.append(var.value.real)
                else:
                    dv.append(var.value)

        # update the aerodynamic mesh
        if self.aero_nnodes > 0:
            self.m.run_option(self.ndv)
            self.m.gp_file(self.aero_gp_file)
            self.m.dv_file(self.dv_file)
            self.m.usd_file(self.usd_file)

            self.aero_sd = np.zeros(3*self.ndv*self.aero_nnodes)
            self.aero_X[:] *= 0.0

            if type(self.aero_X[0]) == np.complex128:
                aero_X_real=self.aero_X[:].real.copy()
                self.m.perturb(dv,aero_X_real,self.aero_id, self.aero_sd)
                self.aero_X = aero_X_real + 0.0j
            else:
                self.m.perturb(dv,self.aero_X,self.aero_id, self.aero_sd)

        # update the structural mesh
        if create_map:
            self.struct_nnodes_total = self.comm.allreduce(self.struct_nnodes)

        if self.struct_nnodes > 0:
            self.m.run_option(self.ndv)
            self.m.gp_file(self.struct_gp_file)
            self.m.dv_file(self.dv_file)
            self.m.usd_file(self.usd_file)

            if create_map:
                self.struct_X = np.zeros(3*self.struct_nnodes_total,dtype=type(self.struct_X[0]))
                self.struct_sd = np.zeros(3*self.ndv*self.struct_nnodes_total)
                self.struct_id = range(self.struct_nnodes_total)
            else:
                self.struct_X = np.zeros(3*self.struct_nnodes,dtype=type(self.struct_X[0]))
                self.struct_sd = np.zeros(3*self.ndv*self.struct_nnodes)
            struct_id = np.array(self.struct_id[:])+1

            if type(self.struct_X[0]) == np.complex128:
                struct_X_real=self.struct_X.real.copy()
                self.m.perturb(dv,struct_X_real,struct_id, self.struct_sd)
                self.struct_X = struct_X_real + 0.0j
            else:
                self.m.perturb(dv,self.struct_X,struct_id, self.struct_sd)


    def _form_struct_map(self,struct_X):
        """
        If no structural global id's are given, form a map to act as id list

        Parameters
        ----------
        struct_X: list of float or numpy.array
            coordinates from analysis tool

        Returns
        -------
        struct_map: numpy.array of type int64
            map from analysis tool to MASSOUD ordering

        """
        struct_map = np.zeros(self.struct_nnodes,dtype='int64')
        os.chdir("./Rubberize")
        dv = np.zeros(self.ndv)
        self._massoud_update(dv,create_map=True)
        for i in xrange(self.struct_nnodes):
            for j in xrange(self.struct_nnodes_total):
                if ( abs(struct_X[3*i  ] - self.struct_X[3*j  ])<1e-8 and
                     abs(struct_X[3*i+1] - self.struct_X[3*j+1])<1e-8 and
                     abs(struct_X[3*i+2] - self.struct_X[3*j+2])<1e-8 ):
                    struct_map[i] = j
                    break
            else:
                print 'Warning struct map NOT SET in massoud body'
        os.chdir("..")
        return struct_map

    def _complex_perturb(self):
        """
        Give a complex perturbation to the mesh
        """

        # Zero out the imaginary part of struct_X
        for i in xrange(self.struct_nnodes):
            self.struct_X[i] = self.struct_X[i].real

        for dv,var in enumerate(self.variables['shape']):
            for i in xrange(self.aero_nnodes):
                self.aero_X[3*i  ] += var.value.imag * 1.0j * self.aero_sd[i*self.ndv*3+dv*3  ]
                self.aero_X[3*i+1] += var.value.imag * 1.0j * self.aero_sd[i*self.ndv*3+dv*3+1]
                self.aero_X[3*i+2] += var.value.imag * 1.0j * self.aero_sd[i*self.ndv*3+dv*3+2]

            for i in xrange(self.struct_nnodes):
                self.struct_X[3*i  ] += var.value.imag * 1.0j * self.struct_sd[i*self.ndv*3+dv*3  ]
                self.struct_X[3*i+1] += var.value.imag * 1.0j * self.struct_sd[i*self.ndv*3+dv*3+1]
                self.struct_X[3*i+2] += var.value.imag * 1.0j * self.struct_sd[i*self.ndv*3+dv*3+2]
